#include<bits/stdc++.h>
using namespace std;

vector<int> UnionofSorted(vector<int>& nums1, vector<int> &nums2){
    int n = nums1.size(), m = nums2.size();
    int i=0,j=0;
    vector<int> a;
    while(i<n && j<m){
        if(a.empty()){
            if(nums1[i]<nums2[j]){
                a.push_back(nums1[i]);
                i++;
            }else{
                a.push_back(nums2[j]);
                j++;
            }
        }
        
        if(nums1[i]<nums2[j]){
            if(nums1[i]!=a.back()){
                a.push_back(nums1[i]);
            }
            i++;
        }else{
            if(j!=0 && nums2[j]!=a.back()){
                a.push_back(nums2[j]);
            }
            j++;
        }
    }
    
    while(i<n){
        if(!a.empty()){
            a.push_back(nums1[i]);
        }else if(nums1[i]!=a.back()){
                a.push_back(nums1[i]);
            }
        i++;
    }
    while(j<m){
        if(!a.empty()){
            a.push_back(nums2[j]);
        }else if(nums2[j]!=a.back()){
                a.push_back(nums2[j]);
            }
        j++;
    }
    return a;
}

vector<int> getarr(){
    string s,num="";
    getline(cin,s);
    vector<int> arr;
    for(char c:s){
        if(isdigit(c) || c=='-'){
            num+=c;
        }else if(!num.empty()){
            arr.push_back(stoi(num));
            num="";
        }
    }
    if(!num.empty()){
        arr.push_back(stoi(num));
    }
    return arr;
}

void printarr(vector<int> &a){
    cout<<"[";
    for(int i=0;i<a.size();i++){
        cout<<a[i]<<", ";
    }
    cout<<"]";
}

int main(){
    
    vector<int>arr1 = getarr();
    vector<int>arr2 = getarr();

    vector<int> arr = UnionofSorted(arr1,arr2);
    printarr(arr);
    return 0;
}
