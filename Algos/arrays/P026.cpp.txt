#include<bits/stdc++.h>
using namespace std;


void setZeroes(vector<vector<int>>& matrix) {
    int m = matrix.size(),n = matrix[0].size();
    bool col_0 = false,row_0 = false;
    for(int i=0;i<m;i++){
        if(!matrix[i][0]) col_0 = true;
    }
    
    for(int i=0;i<n;i++){
        if(!matrix[0][i]) row_0 = true;
    }
    
    for(int i=0;i<m;i++){
        for(int j=0;j<n;j++){
            if(!matrix[i][j]){
                matrix[i][0] = matrix[0][j] = 0;
            }
        }
    }
    for(int i=1;i<m;i++){
        for(int j=1;j<n;j++){
            if(!matrix[i][0] || !matrix[0][j]){
                matrix[i][j] = 0;
            }
        }
    }

    if(col_0){
        for(int i=0;i<m;i++){
            matrix[i][0] = 0;
        }
    }
    
    if(row_0){
        for(int i=0;i<n;i++){
            matrix[0][i] = 0;
        }
    }
}

vector<vector<int>> getmatrix(){
    string s,num="";
    getline(cin,s);
    int m = 0,n = 0,t = 0,N = s.length();
    for(int i=0;i<N;i++){
        if(i==0 || i==N-1) continue;
        else if(s[i]=='['){
            m++;
        }else if(isdigit(s[i]) && !isdigit(s[i+1])){
            t++;
        }
    }
    n = t/m;
    int i=0,j=0;
    vector<vector<int>> matrix(m,vector<int>(n,0));
    for(char c:s){
        if(j>=n){
            j=0;
            i++;
        }
        
        if(isdigit(c) || c=='-'){
            num+=c;
        }else if(!num.empty()){
            matrix[i][j] = stoi(num);
            num="";
            j++;
        }
    }
    if(!num.empty()){
        matrix[i][j] = stoi(num);
    }
    return matrix;
}

void printMat(vector<vector<int>> &mat){
    int m = mat.size(),n = mat[0].size();
    
    cout<<"[";
    for(int i=0;i<m;i++){
        cout<<"[";
        for(int j=0;j<n;j++){
            cout<<mat[i][j];
        }
        if(i==m-1) cout<<"]"; 
        else cout<<"]"<<endl;
    }
    cout<<"]";
}

int main(){
    
    auto mat = getmatrix();

    setZeroes(mat);
    printMat(mat);
    return 0;
}
