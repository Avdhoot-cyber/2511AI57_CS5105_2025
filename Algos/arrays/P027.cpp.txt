#include<bits/stdc++.h>
using namespace std;


void rotate(vector<vector<int>>& matrix) {
    int m = matrix.size(),n = matrix[0].size();
    
    for(int i=0;i<m;i++){
        for(int j=0;j<i;j++){
            swap(matrix[i][j],matrix[j][i]);
        }
    }
    
    for(int i=0;i<m;i++){
        for(int j=0;j<n/2;j++){
            swap(matrix[i][j],matrix[i][n-1-j]);
        }
    }
}

vector<vector<int>> getmatrix(){
    string s,num="";
    getline(cin,s);
    int m = 0,n = 0,t = 0,N = s.length();
    for(int i=0;i<N;i++){
        if(i==0 || i==N-1) continue;
        else if(s[i]=='['){
            m++;
        }else if(isdigit(s[i]) && !isdigit(s[i+1])){
            t++;
        }
    }
    n = t/m;
    int i=0,j=0;
    vector<vector<int>> matrix(m,vector<int>(n,0));
    for(char c:s){
        if(j>=n){
            j=0;
            i++;
        }
        
        if(isdigit(c) || c=='-'){
            num+=c;
        }else if(!num.empty()){
            matrix[i][j] = stoi(num);
            num="";
            j++;
        }
    }
    if(!num.empty()){
        matrix[i][j] = stoi(num);
    }
    return matrix;
}

void printMat(vector<vector<int>> &mat){
    int m = mat.size(),n = mat[0].size();
    
    cout<<"[";
    for(int i=0;i<m;i++){
        cout<<"[";
        for(int j=0;j<n;j++){
            cout<<mat[i][j];
        }
        if(i==m-1) cout<<"]"; 
        else cout<<"]"<<endl;
    }
    cout<<"]";
}

int main(){
    
    auto mat = getmatrix();

    rotate(mat);
    printMat(mat);
    return 0;
}
