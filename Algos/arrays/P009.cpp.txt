#include<bits/stdc++.h>
using namespace std;

int optimalMaxConsOnes(int arr[],int n){
    int max_len = 0,curr_len=0;
    for(int i=0;i<n;i++){
        if(arr[i]==1){
            while(arr[i]==1){
                curr_len+=1;
                i++;
                if(i>=n){
                    max_len = max(max_len,curr_len);
                    break;
                }
            }
            max_len = max(max_len,curr_len);
            curr_len = 0;
        }
    }
    return max_len;    
}

int main(){
    int n;
    cout<<"enter array len:";
    cin>>n;
    int arr[n];
    for(int i=0;i<n;i++){
        cin>>arr[i];
    }
    
    int maxones = optimalMaxConsOnes(arr,n);
    cout<<maxones;
    return 0;
}
