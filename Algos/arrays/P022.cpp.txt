#include<bits/stdc++.h>
using namespace std;

void sort012(vector<int>& nums) {
    int n = nums.size(),c_0 = 0,c_1 = 0,c_2 = 0;
    
    for(int i=0;i<n;i++){
        if(nums[i]==0){
            c_0++;
        }else if(nums[i]==1){
            c_1++;
        }else{
            c_2++;
        }
    }
    int i = 0;
    while(c_0 || c_1 || c_2){
        while(c_0){
            nums[i] = 0;
            i++;
            c_0--;
        }
        while(c_1){
            nums[i] = 1;
            i++;
            c_1--;
        }
        while(c_2){
            nums[i] = 2;
            i++;
            c_2--;
        }
    }
}

vector<int> getarr(){
    string s,num="";
    getline(cin,s);
    vector<int> arr;
    for(char c:s){
        if(isdigit(c) || c=='-'){
            num+=c;
        }else if(!num.empty()){
            arr.push_back(stoi(num));
            num="";
        }
    }
    if(!num.empty()){
        arr.push_back(stoi(num));
    }
    return arr;
}

void printarr(vector<int> &a){
    cout<<"[";
    for(int i=0;i<a.size();i++){
        cout<<a[i]<<", ";
    }
    cout<<"]";
}

int main(){
    
    vector<int>arr = getarr();

    sort012(arr);
    printarr(arr);
    return 0;
}
