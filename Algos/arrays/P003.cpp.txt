#include<bits/stdc++.h>
using namespace std;

bool optimalIsSorted(int arr[],int n){
    for(int i=0;i<n-1;i++){
        if(arr[i]>arr[i+1]){
            return false;
        }
    }
    return true;
}

int main(){
    int n;
    cout<<"enter array len:";
    cin>>n;
    int arr[n];
    for(int i=0;i<n;i++){
        cin>>arr[i];
    }
    bool ans = optimalIsSorted(arr,n);
    ans?cout<<"true":cout<<"false";
    return 0;
}
