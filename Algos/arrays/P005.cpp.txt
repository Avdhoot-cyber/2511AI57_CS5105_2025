#include<bits/stdc++.h>
using namespace std;

void optimalRotLeftbyOne(int arr[],int n){
    int val = arr[0];
    for(int i=0;i<n;i++){
        if(i==n-1){
            arr[i] = val;
        }else{
            arr[i] = arr[i+1];
        }
        
    }
}

int main(){
    int n;
    cout<<"enter array len:";
    cin>>n;
    int arr[n];
    for(int i=0;i<n;i++){
        cin>>arr[i];
    }
    optimalRotLeftbyOne(arr,n);
    for(int i=0;i<n;i++){
        cout<<arr[i]<<" ";
    }
    return 0;
}
