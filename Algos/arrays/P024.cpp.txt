#include<bits/stdc++.h>
using namespace std;

vector<int> rearrangeArray(vector<int>& nums) {
    int pos_index = 0,neg_index = 1,n = nums.size();
    
    vector<int> v(n);
    
    for(int i=0;i<n;i++){
        if(nums[i]>0){
            v[pos_index] = nums[i];
            pos_index+=2;
        }else{
            v[neg_index] = nums[i];
            neg_index+=2;
        }
    }
    return v;
}

vector<int> getarr(){
    string s,num="";
    getline(cin,s);
    vector<int> arr;
    for(char c:s){
        if(isdigit(c) || c=='-'){
            num+=c;
        }else if(!num.empty()){
            arr.push_back(stoi(num));
            num="";
        }
    }
    if(!num.empty()){
        arr.push_back(stoi(num));
    }
    return arr;
}

void printarr(vector<int> &a){
    cout<<"[";
    for(int i=0;i<a.size();i++){
        cout<<a[i]<<", ";
    }
    cout<<"]";
}

int main(){
    
    vector<int>arr = getarr();

    vector<int> ans = rearrangeArray(arr);
    printarr(ans);
    return 0;
}
