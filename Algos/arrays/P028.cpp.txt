#include<bits/stdc++.h>
using namespace std;

void printarr(vector<int> &a){
    cout<<"[";
    for(int i=0;i<a.size();i++){
        cout<<a[i]<<", ";
    }
    cout<<"]";
}

vector<int> spiralOrder(vector<vector<int>>& matrix) {
    int m = matrix.size(), n = matrix[0].size(),index=0;
    vector<int> o(m*n);
    int top = 0, bottom = m - 1, left = 0, right = n - 1;
    while (top <= bottom && left <= right) {
        for (int j = left; j <= right; j++)
            o[index++] = matrix[top][j];
        top++;
    
        for (int i = top; i <= bottom; i++)
            o[index++] = matrix[i][right];
        right--;
    
        if (top <= bottom) {
            for (int j = right; j >= left; j--)
                o[index++] = matrix[bottom][j];
            bottom--;
        }
    
        if (left <= right) {
            for (int i = bottom; i >= top; i--)
                o[index++] = matrix[i][left];
            left++;
        }
    }
    return o;
}

vector<vector<int>> getmatrix(){
    string s,num="";
    getline(cin,s);
    int m = 0,n = 0,t = 0,N = s.length();
    for(int i=0;i<N;i++){
        if(i==0 || i==N-1) continue;
        else if(s[i]=='['){
            m++;
        }else if(isdigit(s[i]) && !isdigit(s[i+1])){
            t++;
        }
    }
    n = t/m;
    int i=0,j=0;
    vector<vector<int>> matrix(m,vector<int>(n,0));
    for(char c:s){
        if(j>=n){
            j=0;
            i++;
        }
        
        if(isdigit(c) || c=='-'){
            num+=c;
        }else if(!num.empty()){
            matrix[i][j] = stoi(num);
            num="";
            j++;
        }
    }
    if(!num.empty()){
        matrix[i][j] = stoi(num);
    }
    return matrix;
}

void printMat(vector<vector<int>> &mat){
    int m = mat.size(),n = mat[0].size();
    
    cout<<"[";
    for(int i=0;i<m;i++){
        cout<<"[";
        for(int j=0;j<n;j++){
            cout<<mat[i][j];
        }
        if(i==m-1) cout<<"]"; 
        else cout<<"]"<<endl;
    }
    cout<<"]";
}



int main(){
    
    auto mat = getmatrix();

    vector<int> arr = spiralOrder(mat);
    printarr(arr);
    return 0;
}
