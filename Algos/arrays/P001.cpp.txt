#include<bits/stdc++.h>
using namespace std;

int optimal(int arr[],int n){
    int largest = INT_MIN;
    for(int i=0;i<n;i++){
        if(arr[i]>largest){
            largest=arr[i];
        }
    }
    return largest;
}

int main(){
    int n;
    cout<<"enter array len:";
    cin>>n;
    int arr[n];
    for(int i=0;i<n;i++){
        cin>>arr[i];
    }
    int largest = optimal(arr,n);
    cout<<largest;
    return 0;
}
