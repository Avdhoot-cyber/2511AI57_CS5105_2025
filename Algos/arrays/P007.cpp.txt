#include<bits/stdc++.h>
using namespace std;

int optimalLinearSearch(int arr[],int n,int target){
    for(int i=0;i<n;i++){
        if(arr[i]==target){
            return i;
        }
    }
    return -1;
}

int main(){
    int n,target;
    cout<<"enter array len:";
    cin>>n;
    int arr[n];
    for(int i=0;i<n;i++){
        cin>>arr[i];
    }
    cout<<"enter target";
    cin>>target;
    int firstOcc = optimalLinearSearch(arr,n,target);
    cout<<firstOcc;
    return 0;
}
