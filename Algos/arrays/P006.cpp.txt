#include<bits/stdc++.h>
using namespace std;

void optimalMoveZeroestoEnd(int arr[],int n){
    int index = 0,i=0;
    for(int i=0;i<n;i++){
        if(arr[i]!=0){
            arr[index] = arr[i];
            index++;
        }
    }
    while(index<n){
        arr[index] = 0;
        index++;
    }
}

int main(){
    int n;
    cout<<"enter array len:";
    cin>>n;
    int arr[n];
    for(int i=0;i<n;i++){
        cin>>arr[i];
    }
    optimalMoveZeroestoEnd(arr,n);
    for(int i=0;i<n;i++){
        cout<<arr[i]<<" ";
    }
    return 0;
}
